<script>

    function analyzeABG() {

        const ph = parseFloat(document.getElementById('ph').value);

        const paco2 = parseFloat(document.getElementById('paco2').value);

        const hco3 = parseFloat(document.getElementById('hco3').value);


        if (isNaN(ph) || isNaN(paco2) || isNaN(hco3)) {

            alert("Vui lòng nhập đầy đủ các thông số pH, PaCO2 và HCO3-!");

            return;

        }


        let title = "", logic = "", treat = "";

        const area = document.getElementById('resultArea');


        if (ph < 7.35) {

            if (paco2 > 45) {

                title = "TOAN HÔ HẤP";

                logic = "Do ứ khí CO2 (Giảm thông khí phổi).";

                treat = "Thông thoáng đường thở, hút đờm, hỗ trợ hô hấp (BiPAP/Máy thở), giải quyết nguyên nhân gây tắc nghẽn.";

            } else if (hco3 < 22) {

                title = "TOAN CHUYỂN HÓA";

                let winter = (1.5 * hco3 + 8).toFixed(1);

                logic = `Mất kiềm hoặc tăng acid. Dự đoán PaCO2 bù (Winter): ${winter} mmHg.`;

                treat = "Bù dịch NaCl 0.9%, tìm nguyên nhân (Sốc, Ceton, Suy thận). Chỉ bù Kiềm nếu pH < 7.1.";

            }

        } else if (ph > 7.45) {

            if (paco2 < 35) {

                title = "KIỀM HÔ HẤP";

                logic = "Do thải CO2 quá mức (Tăng thông khí).";

                treat = "Trấn an, giảm đau, hạ sốt. Nếu do tâm lý, cho thở chậm lại hoặc thở túi giấy.";

            } else if (hco3 > 26) {

                title = "KIỀM CHUYỂN HÓA";

                logic = "Do mất acid (nôn) hoặc hạ Kali máu.";

                treat = "Truyền NaCl 0.9%, bù Kali (KCL), ngừng thuốc lợi tiểu nếu có.";

            }

        } else {

            title = "BÌNH THƯỜNG / CÓ BÙ";

            logic = "Các chỉ số nằm trong ngưỡng an toàn hoặc cơ thể đã bù trừ hoàn toàn.";

            treat = "Tiếp tục theo dõi dấu hiệu sinh tồn và tình trạng lâm sàng.";

        }


        document.getElementById('resTitle').innerText = title;

        document.getElementById('resLogic').innerText = logic;

        document.getElementById('resTreat').innerText = treat;

        area.style.display = "block";

        area.scrollIntoView({ behavior: 'smooth' });

    }

</script>
